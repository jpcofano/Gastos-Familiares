# F9.73 — Fix logos: resolutor comercio→logo vía Brand Search API (encima de F9.67)

> Sigue a F9.67 (que dejó `<MerchantLogo>` + mapa curado). Este prompt **solo cambia el resolutor por dentro**:
> pasa de "mapa curado → dominio" a "Brand Search API → icon". `MerchantLogo` y todos sus usos quedan igual. Un archivo
> nuevo + tocar `MerchantLogo`. Solo frontend, deploy `--only hosting`.

---

## Por qué

El mapa curado sólo cubre comercios que enumeramos a mano. La **Brand Search API** de Brandfetch resuelve
*nombre→logo* dinámicamente. Endpoint (ya tenemos client ID `1idDEHYBi7zAzQv9-MQ`):

`GET https://api.brandfetch.io/v2/search/{query}?c={CLIENT_ID}` → array de candidatos:

```json
[{ "brandId":"idaV6DuajO","claimed":false,"domain":"aysa.com.ar","name":"AySA S.A.",
   "icon":"https://cdn.brandfetch.io/idaV6DuajO/w/128/h/128/fallback/lettermark/icon.webp?c=…",
   "_score":14.81,"qualityScore":0.68,"verified":false }]
```

`icon` ya es una URL de logo lista (con fallback lettermark). Estrategia: **override curado → Search API → monograma**,
sin buscar payees que son nombre de persona (transferencias P2P), con caché en `localStorage`.

## §1 · `src/datos/comerciosLogos.ts` (nuevo — reemplaza el uso de `comerciosDominios.ts` como resolutor)

```ts
const CID = import.meta.env.VITE_BRANDFETCH_CLIENT_ID ?? '1idDEHYBi7zAzQv9-MQ';
const norm = (s: string) => (s ?? '').toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g, '').trim();
const logoCDN = (d: string) => `https://cdn.brandfetch.io/domain/${d}/w/120/h/120?c=${CID}`;

// Overrides curados (ganan sobre la búsqueda; evita ambigüedad). Reusá COMERCIOS_DOMINIOS de F9.67.
import { COMERCIOS_DOMINIOS } from './comerciosDominios';

// ¿parece comercio? (tiene forma societaria o está en overrides). Conservador: si no, no busca (P2P).
function pareceComercio(nombre: string): boolean {
  const n = norm(nombre);
  if (!n) return false;
  if (COMERCIOS_DOMINIOS.some(o => o.match.some(m => n.includes(m)))) return true;
  return /\b(s\.?a\.?|s\.?r\.?l|srl|s\.?a\.?s|inc|ltda|coop|ute)\b/.test(n);
}

const mem = new Map<string, string | null>();
const lsGet = (k: string) => { try { return localStorage.getItem('gf-logo:' + k); } catch { return null; } };
const lsSet = (k: string, v: string) => { try { localStorage.setItem('gf-logo:' + k, v); } catch {} };

export async function logoDeComercio(nombre: string): Promise<string | null> {
  const key = norm(nombre);
  if (!key) return null;
  if (mem.has(key)) return mem.get(key)!;
  const ls = lsGet(key); if (ls !== null) { const v = ls || null; mem.set(key, v); return v; }

  // (a) override curado → Logo CDN por dominio
  const ov = COMERCIOS_DOMINIOS.find(o => o.match.some(m => key.includes(m)));
  if (ov) { const u = logoCDN(ov.dominio); mem.set(key, u); lsSet(key, u); return u; }

  // (b) Brand Search — solo si parece comercio
  if (!pareceComercio(nombre)) { mem.set(key, null); lsSet(key, ''); return null; }
  try {
    const r = await fetch(`https://api.brandfetch.io/v2/search/${encodeURIComponent(nombre)}?c=${CID}`);
    const arr = (await r.json()) as Array<{ icon?: string; domain?: string; verified?: boolean; claimed?: boolean; qualityScore?: number }>;
    const best = (arr || []).filter(x => x.icon || x.domain)
      .sort((a, b) => (Number(b.verified) - Number(a.verified)) || (Number(b.claimed) - Number(a.claimed)) || ((b.qualityScore ?? 0) - (a.qualityScore ?? 0)))[0];
    const url = best?.icon || (best?.domain ? logoCDN(best.domain) : null);
    mem.set(key, url); lsSet(key, url || '');
    return url;
  } catch { mem.set(key, null); return null; }
}
```

## §2 · `MerchantLogo` — pasar a async (resolver con `logoDeComercio`)

Reemplazá la resolución síncrona por dominio de F9.67 por el fetch async cacheado:

```tsx
import { useState, useEffect } from 'react';
import { logoDeComercio } from '../../../datos/comerciosLogos';

export function MerchantLogo({ nombre, size = 30, radius = 8 }: { nombre: string; size?: number; radius?: number }) {
  const [url, setUrl] = useState<string | null>(null);
  const [fail, setFail] = useState(false);
  useEffect(() => { let ok = true; setFail(false); setUrl(null); logoDeComercio(nombre).then(u => ok && setUrl(u)); return () => { ok = false; }; }, [nombre]);
  if (!url || fail) {
    return (
      <span style={{ width: size, height: size, borderRadius: radius, background: 'var(--gf-gray-100)', color: 'var(--gf-gray-500)',
        display: 'inline-flex', alignItems: 'center', justifyContent: 'center', flexShrink: 0, fontSize: Math.round(size * 0.4), fontWeight: 700 }}>
        {(nombre || '?').trim().charAt(0).toUpperCase()}
      </span>
    );
  }
  return <img src={url} alt={nombre} onError={() => setFail(true)}
    style={{ width: size, height: size, borderRadius, objectFit: 'contain', background: '#fff', border: '1px solid var(--gf-gray-150)', flexShrink: 0 }} />;
}
```

Los consumidores (Por descripción, detalle del día, etc.) **no cambian** — siguen usando `<MerchantLogo nombre=… />`.
